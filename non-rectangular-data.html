6.5 Rectangling

Setup code

library(tidyverse)
library(here)

komoot <- readRDS(here("data/komoot.rds"))

komoot_content <-
  komoot %>%
  pull(content)

coordinates_transposed <-
  komoot_content %>%
  map(list("features", 1, "geometry", "coordinates")) %>%
  transpose() %>%
  map(~ flatten_dbl(.))
coordinates_transposed %>%
  set_names(c("lon", "lat"))
## $lon
## [1]  13.388860 -79.387207  34.780527   8.542322
